import numpy as np
import matplotlib.pyplot as plt

# Define parameters
alpha = 1.0
t = np.linspace(0, 10, 1000)  # Time range
f_t = np.exp(-alpha * t)  # Exponential decay function

# Calculate Fourier transform
omega = np.linspace(-10, 10, 1000)
F_omega = 1 / (alpha + 1j * omega)  # Fourier transform result

# Plot results
plt.figure(figsize=(12, 6))

# Plot time-domain signal
plt.subplot(1, 2, 1)
plt.plot(t, f_t)
plt.title('Exponential Decay Function $f(t) = e^{-\\alpha t}$')
plt.xlabel('Time t')
plt.ylabel('f(t)')
plt.grid()

# Plot frequency-domain signal
plt.subplot(1, 2, 2)
plt.plot(omega, np.abs(F_omega), label='Magnitude Spectrum')
plt.plot(omega, np.angle(F_omega), label='Phase Spectrum')
plt.title('Fourier Transform $F(\\omega)$')
plt.xlabel('Frequency $\\omega$')
plt.ylabel('$F(\\omega)$')
plt.legend()
plt.grid()

plt.tight_layout()
plt.show()
